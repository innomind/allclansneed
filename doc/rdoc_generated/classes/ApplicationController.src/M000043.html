  def user_has_right_ge? level
    unless user_is_guest?
      right = UserRight.find :first, :conditions => {:site_id => current_site_id, :user_id => current_user_id}
      if right.nil?
        (level == LEVEL_ACN_MEMBER)
      else
        (right.level >= level)
      end
    else
      false
    end
  end
