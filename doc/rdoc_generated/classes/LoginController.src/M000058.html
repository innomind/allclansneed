  def create
    unless params['user'].nil?
      @user = User.new params[:user]
      @user.user_rights[0] = UserRight.new(:user_id => @user.id, :site_id => params[:site_id], :level => 1)
      if !@user.save
        flash[:errors] = @user.errors
      else
        render :partial => 'create_success'
      end
    end
    @pages = Site.all
  end
