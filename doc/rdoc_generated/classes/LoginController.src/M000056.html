  def login
    nick = params[:nick]
    pw = params[:password]
    unless (nick.nil? || pw.nil?)
      usr = User.first :conditions => {:login => nick}
      if !usr.nil? && usr.check_pw(pw)
        session['user'] = usr
        session['user_sites'] = usr.site_ids
        @logged_in = true
      else
        flash.now[:notice] = 'login failed'+' for "'+params[:nick]+'"'
      end
      
    end
    render :action => 'index'
  end
