  def create_squad
    @squad = Squad.new(params[:squad])
    @squad.clan = @clan
    if @squad.save
      flash[:notice] = 'Clan successfully created.'
    else
      flash[:error] = "Clan couldn't be created"
    end
    redirect_to :controller => 'clan_management'
  end
