  def create
    @_post = params[:news]
    
    @news = News.new( :title => @_post['title'],
                      :subtext => @_post['subtext'],
                      :news => @_post['news'])
                      
    @news.user = current_user
    @news.site = current_site
    @news.tag_list = @_post['tags']
    
    if @news.save
      redirect_to :action => 'index'
    else
      render :action => 'new'
    end
  end
